-- Fun√ß√£o para encontrar jogador
local function findPlayer(partialName)
    partialName = partialName:lower()
    for _, player in pairs(Players:GetPlayers()) do
        if player.Name:lower():find(partialName) or player.DisplayName:lower():find(partialName) then
            return player
        end
    end
    return nil
end

-- Fun√ß√£o para enviar mensagem no chat normal do servidor
local function sendChatMessage(message)
    local success = false
    
    -- M√©todo 1: ReplicatedStorage ChatEvents (Mais comum)
    pcall(function()
        local ReplicatedStorage = game:GetService("ReplicatedStorage")
        local chatEvents = ReplicatedStorage:WaitForChild("DefaultChatSystemChatEvents", 2)
        if chatEvents then
            local sayMessageRequest = chatEvents:WaitForChild("SayMessageRequest", 2)
            if sayMessageRequest then
                sayMessageRequest:FireServer(message, "All")
                success = true
                print("M√©todo 1 funcionou - ReplicatedStorage")
            end
        end
    end)
    
    if not success then
        -- M√©todo 2: TextChatService (Novo sistema de chat)
        pcall(function()
            local TextChatService = game:GetService("TextChatService")
            local textChannel = TextChatService:FindFirstChild("TextChannels")
            if textChannel then
                local rbxGeneral = textChannel:FindFirstChild("RBXGeneral")
                if rbxGeneral then
                    rbxGeneral:SendAsync(message)
                    success = true
                    print("M√©todo 2 funcionou - TextChatService")
                end
            end
        end)
    end
    
    if not success then
        -- M√©todo 3: Chat Service direto
        pcall(function()
            if LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("Head") then
                local ChatService = game:GetService("Chat")
                ChatService:Chat(LocalPlayer.Character.Head, message)
                success = true
                print("M√©todo 3 funcionou - ChatService")
            end
        end)
    end
    
    if not success then
        -- M√©todo 4: StarterPlayer ChatScript
        pcall(function()
            local StarterPlayer = game:GetService("StarterPlayer")
            local chatScript = StarterPlayer:FindFirstChild("StarterPlayerScripts")
            if chatScript then
                local chatMain = chatScript:FindFirstChild("ChatScript")
                if chatMain then
                    local messageData = {
                        Message = message,
                        FromSpeaker = LocalPlayer.Name
                    }
                    -- Tentar enviar via ChatScript
                    success = true
                    print("M√©todo 4 funcionou - ChatScript")
                end
            end
        end)
    end
    
    -- M√©todo 5: RemoteEvent customizado (para alguns servidores)
    if not success then
        pcall(function()
            for _, obj in pairs(game:GetDescendants()) do
                if obj.Name == "SayMessageRequest" and obj:IsA("RemoteEvent") then
                    obj:FireServer(message, "All")
                    success = true
                    print("M√©todo 5 funcionou - RemoteEvent customizado")
                    break
                end
            end
        end)
    end
    
    return success
end

-- Fun√ß√£o para iniciar c√≥pia de jogador espec√≠fico
local function startCopying()
    if PlayerTextBox.Text == "" then
        StatusLabel.Text = "Status: Digite um nick!"
        return
    end
    
    -- Parar modo All se estiver ativo
    if copyAllEnabled then
        stopAllCopying()
    end
    
    targetPlayer = findPlayer(PlayerTextBox.Text)
    if not targetPlayer then
        StatusLabel.Text = "Status: Jogador n√£o encontrado!"
        return
    end
    
    copyingEnabled = true
    StatusLabel.Text = "Status: Copiando " .. targetPlayer.Name
    
    -- Conectar ao chat
    if chatConnection then
        chatConnection:Disconnect()
    end
    
    chatConnection = targetPlayer.Chatted:Connect(function(message)
        if copyingEnabled and targetPlayer then
            print("üì¢ " .. targetPlayer.Name .. " disse: " .. message)
            
            -- Sem delay - instant√¢neo
            local success = sendChatMessage(message)
            
            if success then
                print("‚úÖ Mensagem copiada para o chat: " .. message)
                StatusLabel.Text = "Status: ‚úÖ √öltima: " .. message:sub(1, 20) .. (message:len() > 20 and "..." or "")
            else
                print("‚ùå Falha ao copiar mensagem")
                StatusLabel.Text = "Status: ‚ùå Erro - Verifique o console"
            end
        end
    end)
    
    -- Mensagem de confirma√ß√£o
    print("üéØ Iniciando c√≥pia de: " .. targetPlayer.Name)
    print("üí¨ Aguardando mensagens no chat...")
end

-- Fun√ß√£o para iniciar c√≥pia de TODOS os jogadores
local function startAllCopying()
    -- Parar c√≥pia individual se estiver ativa
    if copyingEnabled then
        stopCopying()
    end
    
    copyAllEnabled = true
    StatusLabel.Text = "Status: üåç Copiando TODO o servidor"
    
    -- Limpar conex√µes anteriores
    if allChatConnection then
        allChatConnection:Disconnect()
    end
    
    -- Conectar a TODOS os jogadores atuais
    local connections = {}
    
    for _, player in pairs(Players:GetPlayers()) do
        if player ~= LocalPlayer then -- N√£o copiar pr√≥prias mensagens
            local connection = player.Chatted:Connect(function(message)
                if copyAllEnabled then
                    print("üåç " .. player.Name .. " disse: " .. message)
                    
                    -- Sem delay - instant√¢neo
                    local success = sendChatMessage(message)
                    
                    if success then
                        print("‚úÖ Copiado de " .. player.Name .. ": " .. message)
                        StatusLabel.Text = "Status: üåç [" .. player.Name .. "]: " .. message:sub(1, 15) .. (message:len() > 15 and "..." or "")
                    else
                        print("‚ùå Falha ao copiar de " .. player.Name)
                    end
                end
            end)
            table.insert(connections, connection)
        end
    end
    
    -- Conectar a novos jogadores que entrarem
    local playerAddedConnection = Players.PlayerAdded:Connect(function(player)
        if copyAllEnabled and player ~= LocalPlayer then
            local connection = player.Chatted:Connect(function(message)
                if copyAllEnabled then
                    print("üåç " .. player.Name .. " disse: " .. message)
                    local success = sendChatMessage(message)
                    if success then
                        StatusLabel.Text = "Status: üåç [" .. player.Name .. "]: " .. message:sub(1, 15) .. (message:len() > 15 and "..." or "")
                    end
                end
            end)
            table.insert(connections, connection)
        end
    end)
    
    -- Armazenar todas as conex√µes
    allChatConnection = {
        connections = connections,
        playerAddedConnection = playerAddedConnection
    }
    
    print("üåç Iniciando c√≥pia de TODO o servidor!")
    print("üí¨ Copiando mensagens de " .. (#connections) .. " jogadores...")
end

-- Fun√ß√£o para parar c√≥pia individual
local function stopCopying()
    copyingEnabled = false
    targetPlayer = nil
    if chatConnection then
        chatConnection:Disconnect()
        chatConnection = nil
    end
end

-- Fun√ß√£o para parar c√≥pia de todos
local function stopAllCopying()
    copyAllEnabled = false
    if allChatConnection then
        -- Desconectar todas as conex√µes
        for _, connection in pairs(allChatConnection.connections) do
            connection:Disconnect()
        end
        if allChatConnection.playerAddedConnection then
            allChatConnection.playerAddedConnection:Disconnect()
        end
        allChatConnection = nil
    end
end

-- Fun√ß√£o para parar tudo
local function stopAll()
    stopCopying()
    stopAllCopying()
    StatusLabel.Text = "Status: Parado"
    print("‚èπÔ∏è Todas as c√≥pias foram paradas")
end

-- Efeitos visuais dos bot√µes
local function createButtonEffect(button, hoverColor, clickColor)
    local originalColor = button.BackgroundColor3
    
    button.MouseEnter:Connect(function()
        if not copyingEnabled or button ~= CopyButton then
            local tween = TweenService:Create(button, TweenInfo.new(0.2), {BackgroundColor3 = hoverColor})
            tween:Play()
        end
    end)
    
    button.MouseLeave:Connect(function()
        if not copyingEnabled or button ~= CopyButton then
            local tween = TweenService:Create(button, TweenInfo.new(0.2), {BackgroundColor3 = originalColor})
            tween:Play()
        end
    end)
    
    button.MouseButton1Down:Connect(function()
        local tween = TweenService:Create(button, TweenInfo.new(0.1), {BackgroundColor3 = clickColor})
        tween:Play()
    end)
    
    button.MouseButton1Up:Connect(function()
        local tween = TweenService:Create(button, TweenInfo.new(0.1), {BackgroundColor3 = originalColor})
        tween:Play()
    end)
end

-- Aplicar efeitos
createButtonEffect(CopyButton, Color3.fromRGB(0, 200, 0), Color3.fromRGB(0, 140, 0))
